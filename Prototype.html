    <script>

        // --- Simulation Constants (Physical & Economic Benchmarks) ---

        const C_P = 900;          // Specific heat of Aluminum (J/kg*K)

        const L_F = 395000;       // Latent heat of fusion (J/kg)

        const T_INITIAL = 293.15; // Initial temp (20°C in K)

        const T_MELT = 933.15;    // Melt temp (660°C in K)

        const SOLAR_IRRADIANCE = 1361; // Solar Irradiance in LEO (W/m^2)

        const J_TO_KWH = 3.6 * 10**6; // Conversion factor: Joules to kWh


        // Benchmarks for Visualization

        const BENCHMARK_LAUNCH_COST = 1500; // $/kg (Conservative LEO launch cost)

        const BENCHMARK_TERRESTRIAL_ENERGY_KWH = 2.78; // kWh/kg (Terrestrial secondary recycling, using 10 MJ/kg)


        // --- DOM Elements ---

        const massInput = document.getElementById('massInput');

        const efficiencyInput = document.getElementById('efficiencyInput');

        const areaInput = document.getElementById('areaInput');


        const massValue = document.getElementById('massValue');

        const efficiencyValue = document.getElementById('efficiencyValue');

        const areaValue = document.getElementById('areaValue');


        const totalEnergyOutput = document.getElementById('totalEnergyOutput');

        const processTimeOutput = document.getElementById('processTimeOutput');

        const energyPerKgOutput = document.getElementById('energyPerKgOutput');

        const costFactorOutput = document.getElementById('costFactorOutput');


        const misarEnergyBar = document.getElementById('misarEnergyBar');

        const terrestrialEnergyBar = document.getElementById('terrestrialEnergyBar');

        const misarEnergyBarText = document.getElementById('misarEnergyBarText');

        

        const recyclingCostBar = document.getElementById('recyclingCostBar');

        const launchCostBar = document.getElementById('launchCostBar');



        function calculateMISARMetrics() {

            // 1. Get current parameters

            const M = parseFloat(massInput.value); // Mass in kg

            const eta_therm = parseFloat(efficiencyInput.value) / 100; // Efficiency (0.1 to 0.9)

            const A = parseFloat(areaInput.value); // Area in m^2


            // Update UI slider values

            massValue.textContent = M.toFixed(0);

            efficiencyValue.textContent = (eta_therm * 100).toFixed(0);

            areaValue.textContent = A.toFixed(0);


            // 2. Calculate Theoretical Energy (in Joules)

            // Q_sensible = M * c_p * delta_T

            const Q_sensible = M * C_P * (T_MELT - T_INITIAL);

            // Q_latent = M * L_f

            const Q_latent = M * L_F;


            const Q_theoretical = Q_sensible + Q_latent; // Total Joules needed


            // 3. Calculate Actual Energy Input (accounting for losses)

            const E_process_Joules = Q_theoretical / eta_therm;

            const E_process_kWh = E_process_Joules / J_TO_KWH; // Total kWh


            // 4. Calculate Energy per Kilogram (The core MISAR efficiency metric)

            const E_misar_per_kg_kWh = E_process_kWh / M;


            // 5. Calculate Processing Time

            // Power absorbed: P_solar = I_solar * A * eta_therm (Watts)

            const P_solar_Watts = SOLAR_IRRADIANCE * A * eta_therm; 


            // Time = Energy / Power (seconds)

            const T_seconds = E_process_Joules / P_solar_Watts;

            const T_hours = T_seconds / 3600;


            // 6. Update Output Display

            totalEnergyOutput.textContent = `${E_process_kWh.toFixed(2)} kWh`;

            processTimeOutput.textContent = `${T_hours.toFixed(2)} hr`;

            energyPerKgOutput.textContent = `${E_misar_per_kg_kWh.toFixed(2)} kWh/kg`;


            // 7. Visualization Updates


            // A. Energy Bar Comparison (MISAR vs Terrestrial Secondary Recycling)

            const misarEnergyRatio = E_misar_per_kg_kWh / BENCHMARK_TERRESTRIAL_ENERGY_KWH;

            

            // Normalize bars based on the highest energy value (Terrestrial in this case)

            const misarBarWidth = Math.max(1, misarEnergyRatio * 100); 

            const terrestrialBarWidth = 100;


            misarEnergyBar.style.width = `${misarBarWidth.toFixed(0)}%`;

            terrestrialEnergyBar.style.width = `${terrestrialBarWidth.toFixed(0)}%`;

            misarEnergyBarText.textContent = `${E_misar_per_kg_kWh.toFixed(2)} kWh/kg`;

            

            // B. Cost Bar Comparison (MISAR vs Launch)

            // MISAR Cost per kg = $0.00

            const misarCostPerKg = 0.00; 


            // Bar widths: Launch is 100%, MISAR is min 1% to be visible

            launchCostBar.style.width = '100%';

            recyclingCostBar.style.width = '1%';

            

            // C. Cost Differential Factor

            if (misarCostPerKg === 0) {

                costFactorOutput.textContent = 'Orders of Magnitude Less';

            } else {

                const costFactor = BENCHMARK_LAUNCH_COST / misarCostPerKg;

                costFactorOutput.textContent = `${costFactor.toFixed(0)}X CHEAPER`;

            }

        }


        // Event Listeners

        massInput.addEventListener('input', calculateMISARMetrics);

        efficiencyInput.addEventListener('input', calculateMISARMetrics);

        areaInput.addEventListener('input', calculateMISARMetrics);


        // Initial run

        calculateMISARMetrics();

    </script>
